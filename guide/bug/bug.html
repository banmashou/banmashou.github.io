内网：不能正常显示地图

解决：

由于原生 MapboxGL 只支持 3857 的坐标系，因此 SuperMap iClient 产品团队在原生 MapboxGL 基础上扩展了一个增强库 mapbox-gl-enhance.js，目前增强库只支持对接 EPSG:4326、EPSG:4490、EPSG:4214、EPSG:4610、EPSG:3857 这几种坐标系的地图。

1）引入 js，以上坐标系的地图除了加载 3857 坐标系的地图外其他的都需要引入增强库 mapbox-gl-enhance.js EPSG:3857 地图：

<script type="text/javascript" src="/fire_system_hp/mapboxgl/iclient-mapboxgl-es6.min.js"></、script>

2）EPSG:4326、EPSG:4490、EPSG:4214、EPSG:4610 的地图必须设置 include 属性：

<script type="text/javascript" include="mapbox-gl-enhance" exclude='iclient9-mapboxgl' src="/fire_system_hp/mapboxgl/iclient-mapboxgl-es6.min.js"></>

3）加载地图

var host = window.isLocal ? window.server : "https://xxx.xx.x.xx:9999";
var url = host + "/iserver/services/map-ugcv5-19HPmapP/rest/maps/19HPmapP";
var attribution =
  "<a href='https://www.mapbox.com/about/maps/' target='_blank'>© Mapbox </a>" +
  " with <span>© <a href='http://iclient.supermap.io' target='_blank'>SuperMap iClient</a> | </span>" +
  " Map Data <span>© <a href='http://support.supermap.com.cn/product/iServer.aspx' target='_blank'>SuperMap iServer</a></span> ";

var map = new mapboxgl.Map({
  container: "map", // container id
  style: {
    version: 8,
    sources: {
      "raster-tiles": {
        attribution: attribution,
        type: "raster",
        // 参数列表：iserver tileimage地图服务地址（到地图名)必填: rasterSource为"iserver"；
        // transparent:可选，默认为true
        tiles: [url],
        rasterSource: "iserver",
        tileSize: 256,
      },
    },
    layers: [
      {
        id: "simple-tiles",
        type: "raster",
        source: "raster-tiles",
        minzoom: 0,
        maxzoom: 18,
      },
    ],
  },
  // leaflet，openlayers默认支持的坐标系都只有3857和4326，其余坐标系均需要自定义
  crs: "EPSG:4490",
  center: [121.489851, 31.21519], // starting position
  zoom: 15, // starting zoom
});
map.addControl(new mapboxgl.NavigationControl(), "top-left");

# 手写 Vue3 v-model

<!-- 父组件 -->
<template>
	<div class="lesson">
		<!-- 原生写法 -->
		<BmInput :value="title" @update:value="change" />
		<!-- v-model 完整写法 -->
		<BmInput v-model:modelValue="title" />
		<!-- v-model 简写 注意：只有 v-model:modelValue 才能简写 -->
		<BmInput v-model="title" />
	</div>
	<hr />
	{{ title }}
</template>

<script>
import BmInput from './components/BmInput.vue'
export default {
	components: { BmInput },
	data() {
		return {
			db,
			title: '斑马兽'
		}
	},
	methods: {
		show(id) {
			const index = this.db.findIndex(l => l.id == id)
			this.db.splice(index, 1)
		},
		// 将子组件传过来的值赋值给 data 里面的 title
		change(v) {
			this.title = v
		}
	}
}
</script>

<style lang="scss" scoped>
.lesson {

	column-gap: 20px;

<!-- 子组件  -->
<template>
	<input type="text" :value="content" @input="change" />
</template>

<script>
export default {
	// 注册父组件值
	props: ['modelValue'],
	// 注册父组件事件
	emits: ['update:modelValue'],
	data() {
		return {
			content: this.modelValue
		}
	},
	methods: {
		change(event) {
			this.content = event.target.value
			// 调用父组件方法
			this.$emit('update:modelValue', this.content)
		}
	}
}
</script>
